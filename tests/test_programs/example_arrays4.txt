fun reverse(arr, keys) {
    var new_arr = array(5);

    var i = 4;
    var j = 0;
    while i > 0 do {
        set(new_arr, j, get(arr, get(keys, i)));
        i = i - 1;
        j = j + 1;
    }

    return new_arr;
}

fun fill_keys(arr) {
    var i = -1;
    while i < 4 do {
        set(arr, i + 1, i + 1);
        i = i + 1;
    }

    return arr;
}

var a1 = array(5);
var a_keys1 = array(5);

a1 = fill_keys(a1);
a_keys1 = a1;

var reversed = array(5);
reversed = reverse(a1, a_keys1);

var i = 0;
while i < 5 do { 
    print_int(get(reversed, i));
