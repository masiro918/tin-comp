var s1 = "Hello world!";
var s2 = "Bye world!";

var i: Int = 0;

while i < 10 do {
	if i % 2 == 0 then {
		print_str(s1);
		i = i + 1;
		continue;
	} else {
		if str_cmp(s1, s2) then {
			// never here
		} else {
			var s = str_cat(s1, s2);
			print_str(s);
		}
	}
	i = i + 1;
}

print_int(i);

var s3 = "2345";
i = str_to_int(s3);

s2 = s1;

print_int(i);
print_str(s2);

var s4: Int = int_to_str(i);
print_str(s4);
